      #3
    \end{column} %--------------------------- left.

      #4

\begin{frame}{Introdução}
\protect\hypertarget{introduuxe7uxe3o}{}
\begin{itemize}
\tightlist
\item
  O \textbf{avanço computacional} gerou um interesse e possibilidade de
  trabalhar com \textbf{inferência} estatística usando
  \textbf{métodos computacionalmente intensivos}.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Estes métodos se baseiam em \textbf{reamostragem} e/ou
  \textbf{simulação}.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  São úteis quando as abordagens tradicionais não são adequadas por
  motivos como:

  \begin{itemize}
  \tightlist
  \item
    Pequenas amostras.
  \item
    Violação de pressupostos.
  \item
    Problemas ``novos'' para os quais não foi desenvolvido um
    procedimento.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Introdução}
\protect\hypertarget{introduuxe7uxe3o-1}{}
\begin{itemize}
\tightlist
\item
  \textbf{Reamostrar} significa \textbf{amostrar várias vezes} uma série
  de valores a partir dos dados observados.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Dois procedimentos importantes são baseados em reamostragem:
  \textbf{Bootstrap} e o \textbf{Teste de Permutação}.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  O \textbf{Bootstrap} é uma interessante ferramenta que permite
  \textbf{construir uma distribuição amostral} e usar esta distribuição
  para fazer inferência.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Os \textbf{Testes de Permutação} tem sua principal aplicação em
  problemas que envolvem
  \textbf{testes de hipóteses de comparação entre grupos} (diferença
  entre médias, proporções, variâncias, correlações, etc). Permite
  construir a distribuição sob \(H_0\).
\end{itemize}
\end{frame}

\hypertarget{bootstrap}{%
\section{Bootstrap}\label{bootstrap}}

\begin{frame}{Bootstrap}
\protect\hypertarget{bootstrap-1}{}
\begin{itemize}
\tightlist
\item
  A ideia do Bootstrap é
  \textbf{simular o processo do paradigma frequentista} porém fazendo
  uso de uma única amostra.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Ao invés de coletar diversas amostras, geramos reamostras da amostra
  original de mesmo tamanho, tomada com reposição.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Para cada amostra, calcula-se a quantidade de interesse, gera-se a
  distribuição amostral e usa esta distribuição para fazer inferência
  (obter estimativas pontuais, intervalares, testar hipóteses).
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Não se baseia em teoremas ou outras suposições distribucionais.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  O número de repetições é arbitrário, mas quanto mais iterações forem
  feitas, mais precisas são as estimativas pontuais e intervalares.
\end{itemize}
\end{frame}

\begin{frame}{Bootstrap}
\protect\hypertarget{bootstrap-2}{}
\begin{itemize}
\tightlist
\item
  A técnica ganhou popularidade com os artigos e livros de Bradley
  Efron.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Quando proposto, enfrentou considerável ceticismo.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  O método não compensa pequenos tamanho de amostra, não cria novos
  dados nem preenche lacunas. Ele meramente parte do pressuposto que a
  amostra original é representativa, e gera novas amostras com
  variações.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Na Estatística moderna, Bootstrap tem um papel importante para avaliar
  propriedades de estimadores.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Na modelagem, bootstrap pode ser usado gerando conjuntos de dados como
  um todo, basta reamostrar as linhas.
\end{itemize}
\end{frame}

\hypertarget{testes-de-permutauxe7uxe3oaleatorizados}{%
\section{Testes de
permutação/aleatorizados}\label{testes-de-permutauxe7uxe3oaleatorizados}}

\begin{frame}{Testes de permutação/aleatorizados}
\protect\hypertarget{testes-de-permutauxe7uxe3oaleatorizados-1}{}
\begin{itemize}
\tightlist
\item
  Podem ser usados para
  \textbf{testar hipóteses, geralmente envolvendo grupos}.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  O procedimento consiste em combinar duas ou mais amostras e
  aleatoriamente realocar as observações em reamostras.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Pode ser feito de forma exaustiva, isto é, obter todas as possíveis
  combinações.
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
\tightlist
\item
  Os testes de permutação exaustiva são práticos para tamanhos de
  amostra relativamente pequenos e são chamados de testes exatos. Para
  tamanhos grandes de amostra se tornam inviáveis
\end{itemize}
\end{frame}

\begin{frame}{Testes de permutação/aleatorizados}
\protect\hypertarget{testes-de-permutauxe7uxe3oaleatorizados-2}{}
De modo geral:

\begin{enumerate}
\tightlist
\item
  Gere um único grupo com as observações de todos os grupos.
\item
  Extraia uma amostra de tamanho igual ao tamanho do grupo A.
\item
  Dos dados restantes, faça o mesmo procedimento para o grupo B.
\item
  Se houver mais grupos, repita o procedimento.
\item
  Calcule a estatística de interesse baseada nos dados da reamostra (ex:
  diferença entre proporções do grupo A e B) e registre.
\item
  Repita o procedimento e construa uma distribuição de probabilidades
  empírica. Esta é a distribuição sob hipótese nula.
\item
  Veja onde o valor observado se encontra nesta distribuição. Se o valor
  observado se encontrar na área mais densa da distribuição, não há
  evidência de diferença, do contrário há.
\end{enumerate}
\end{frame}
